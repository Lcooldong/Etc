void crossLED()
{
   DDRA = 0xff; 
   
   while(1)
   {
	 PORTA = 0x55;
	 _delay_ms(200);
	 PORTA = 0xaa;
        _delay_ms(200);
    }
 }

void stairLED()
 {
    DDRA = 0xff; 
    
   while(1)
   {
      int i;
      PORTA = 0x01;
      _delay_ms(100);
      for(i= 0; i <7; i++)
      {
	PORTA = PORTA << 1;
        _delay_ms(100); 
      }
   }
}
 
 void repeatLED()
{
    DDRA = 0xff; 
 
   while(1)
   {
     int i, j;
      
     PORTA = 0x01;
     for(i = 0; i <7; i++)
     {
	_delay_ms(50); 
	PORTA = PORTA << 1;
     }
    for(j= 0; j <7; j++)
    {
       _delay_ms(50); 
	PORTA = PORTA >>1;
     }
   }
 }

void repeatLED2()
 {
   unsigned char dir = 1;
   DDRA = 0xff;
   PORTA = 0x01;
   
   while(1)
   {
      _delay_ms(100);
      if(dir)
	 PORTA = PORTA << 1;
      else
	 PORTA = PORTA >>1;
      
      if (PORTA == 0x80)
	 dir = 0;
      else if(PORTA == 0x01)
	 dir = 1;
    }
 }
